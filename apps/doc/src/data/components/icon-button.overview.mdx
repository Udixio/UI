import { CodePreview } from '@/components/CodePreview.js';
import { faPlus, faStar as fasStar } from '@fortawesome/free-solid-svg-icons';
import { faStar as farStar } from '@fortawesome/free-regular-svg-icons';

## Usage

Import IconButton from @udixio/ui-react when an action is best represented by an icon.

```bash
import { IconButton } from "@udixio/ui-react"
```

<CodePreview client:load
             code={`
  <div className="flex items-center gap-3">
    <IconButton ariaLabel="Add item" icon={faPlus} onClick={() => console.log('add')} />
  </div>`}
             scope={{ faPlus }}
/>

## Shapes and widths

Control the button outline with `shape` and surface padding with `width`.

<CodePreview client:load
             code={`
  <div className="flex items-center gap-3">
    <IconButton ariaLabel="Rounded" icon={faPlus} shape="rounded" />
    <IconButton ariaLabel="Squared" icon={faPlus} shape="squared" />
    <IconButton ariaLabel="Narrow" icon={faPlus} width="narrow" />
    <IconButton ariaLabel="Wide" icon={faPlus} width="wide" />
  </div>`}
             scope={{ faPlus }}
/>

## States

Use `disabled` to prevent user interaction and `activated` for a persistent selected state.

<CodePreview client:load
             code={`
  <div className="flex items-center gap-3">
    <IconButton ariaLabel="Active" icon={faPlus} activated />
    <IconButton ariaLabel="Disabled" icon={faPlus} disabled />
    <IconButton ariaLabel="Disabled active" icon={faPlus} disabled activated />
  </div>`}
             scope={{ faPlus }}
/>

## Toggle state

Use \`onToggle\` and \`iconSelected\` to turn it into a toggle button.

<CodePreview client:load
             code={`
  <IconButton
    ariaLabel="Favorite"
    icon={farStar}
    iconSelected={fasStar}
    onToggle={(active) => console.log('active:', active)}
  />`}
             scope={{ farStar, fasStar }}
/>

## Link or action

If the `href` prop is provided, it renders an anchor; otherwise it renders a button.

<CodePreview client:load
             code={`
  <div className="flex items-center gap-3">
    <IconButton ariaLabel="Go to Udixio" icon={faPlus} href="https://udixio.dev" />
    <IconButton ariaLabel="Run action" icon={faPlus} onClick={() => console.log('clicked')} />
  </div>`}
             scope={{ faPlus }}
/>
