import { CodePreview } from '@/components/CodePreview.js';
import { iAdd } from '@udixio/icons-rounded-400/add';
import { iStar } from '@udixio/icons-rounded-400/star';
import { iStarFilled } from '@udixio/icons-rounded-400/filled/star';

## Usage

Import IconButton from @udixio/ui-react when an action is best represented by an icon.

```bash
import { IconButton } from "@udixio/ui-react"
```

<CodePreview
  client:load
  code={`
  <div className="flex items-center gap-3">
    <IconButton label="Add item" icon={iAdd} onClick={() => console.log('add')} />
  </div>`}
  scope={{ iAdd }}
/>

## Toggle state

Use \`onToggle\` and \`iconSelected\` to turn it into a toggle button.

<CodePreview
  client:load
  code={`
  <div className="flex items-center gap-3">
    <IconButton
      label="Standard"
      icon={iStar}
      iconSelected={iStarFilled}
      onToggle={(active) => console.log('active:', active)}
    />
    <IconButton
      variant="filled"
      label="Filled"
      icon={iStar}
      iconSelected={iStarFilled}
      onToggle={(active) => console.log('active:', active)}
    />
    <IconButton
      variant="tonal"
      label="Tonal"
      icon={iStar}
      iconSelected={iStarFilled}
      onToggle={(active) => console.log('active:', active)}
    />
    <IconButton
      variant="outlined"
      label="Outlined"
      icon={iStar}
      iconSelected={iStarFilled}
      onToggle={(active) => console.log('active:', active)}
    />
  </div>
  `}
  scope={{ iStar, iStarFilled }}
/>

## Link or action

If the `href` prop is provided, it renders an anchor; otherwise it renders a button.

<CodePreview
  client:load
  code={`
  <div className="flex items-center gap-3">
    <IconButton label="Go to Udixio" icon={iAdd} href="https://udixio.dev" />
    <IconButton label="Run action" icon={iAdd} onClick={() => console.log('clicked')} />
  </div>`}
  scope={{ iAdd }}
/>
