import { CodePreview } from '@/components/CodePreview.js';
import { iEdit } from '@udixio/icons-rounded-400/edit';
import { iAdd } from '@udixio/icons-rounded-400/add';
import { iShare } from '@udixio/icons-rounded-400/share';

## Usage

Floating Action Button (Fab) represents the primary action on a screen. Import it from @udixio/ui-react.

```bash
import { Fab } from "@udixio/ui-react"
```

<CodePreview client:load
             code={`
  <div className="flex items-end gap-4">
    <Fab label="Create" icon={iAdd} />
    <Fab label="Create" icon={iAdd} isExtended />
  </div>`}
             scope={{ iAdd }}
/>

## Variants

- primary (default)
- secondary
- tertiary
- surface

<CodePreview client:load
             code={`
  <div className="flex items-end gap-4">
    <Fab label="Primary" icon={iAdd} variant="primary" />
    <Fab label="Secondary" icon={iAdd} variant="secondary" />
    <Fab label="Tertiary" icon={iAdd} variant="tertiary" />
    <Fab label="Surface" icon={iAdd} variant="surface" />
  </div>`}
             scope={{ iAdd }}
/>

## Sizes

Available sizes: small, medium (default), large.

<CodePreview client:load
             code={`
  <div className="flex items-end gap-4">
    <Fab label="Edit" icon={iEdit} size="small" />
    <Fab label="Edit" icon={iEdit} size="medium" />
    <Fab label="Edit" icon={iEdit} size="large" />
  </div>`}
             scope={{ iEdit }}
/>

## Extended vs. regular

Use `isExtended` to show the label next to the icon.

<CodePreview client:load
             code={`
  <div className="flex items-end gap-4">
    <Fab label="Share" icon={iShare} />
    <Fab label="Share" icon={iShare} isExtended />
  </div>`}
             scope={{ iShare }}
/>

## Link or action

If `href` is provided, the component renders an anchor; otherwise it renders a button. The `label` is used as aria-label for the non-extended Fab.

<CodePreview client:load
             code={`
  <Fab label="Create" icon={iAdd} href="https://udixio.dev" />
`}
             scope={{ iAdd }}
/>
